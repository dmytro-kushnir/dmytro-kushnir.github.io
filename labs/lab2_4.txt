		<script> 
			"use strict";
			document.write("<p>Використання оператора циклу	<b>FOR</b></p>");			
			// Зведення числа x у ступінь
			let x = parseInt(prompt("Введіть число x")); // задане число
			let y = parseInt(prompt("Введіть ступінь y")); // ступінь, до якого треба звести число
			let z = x; // результат
			
			for (let i = 2; i <= Math.abs(y); i++) z = z * x; z = y > 0? z: 1/z;
			document.write("Число ", x, " в ступені ", y, " дорівнює ", z);

			document.write("<p>Використання оператора циклу <b>WHILE</b></p>");
			let z1 = x; let i = 2;
			while (i <= Math.abs(y)) { 
				z1 = z1 * x;
				i++;
			}
			z1 = y > 0? z1: 1/z1;
			document.write("Число ", x, " в ступені ", y, " дорівнює ", z1);

			document.write("<p>Використання оператора циклу <b>DO WHILE</b></p>");
			let z2 = x; i = 2;
			do {
				z2 = z2 * x; i++;
			} 
			while (i <= Math.abs(y)); z2 = y> 0? z2: 1/z2;
			document.write("Число ", x, " в ступені ", y, " дорівнює ", z2);
		</script>
